pub macro_rules! read_u32 {
    ($elf:ident) => (
        match $elf.ehdr.data {
            types::ELFDATA2LSB => { $elf.file.read_le_u32() }
            types::ELFDATA2MSB => { $elf.file.read_be_u32() }
            types::ELFDATANONE => { panic!("Unable to resolve file endianness"); }
            _ => { panic!("Unable to resolve file endianness"); }
        }
    );
}
